        <div class="field">
            <label>2. Describe your vision...</label>
            <textarea id="promptInput" placeholder="e.g., Zoom in slowly and make it black and white..."></textarea>
        </div>
        <button id="convertBtn">Generate Video</button>
        <div id="status" style="margin-top:15px; color:#4285f4;"></div>
        <video id="outputVideo" controls></video>
        <canvas id="canvas"></canvas>
    </div>
    <div id="subtitleOverlay"></div>

    <script>
        const imageInput = document.getElementById('imageInput');
        const promptInput = document.getElementById('promptInput');
        const convertBtn = document.getElementById('convertBtn');
        const status = document.getElementById('status');
        const video = document.getElementById('outputVideo');
        const canvas = document.getElementById('canvas');
        const subtitles = document.getElementById('subtitleOverlay');
        const ctx = canvas.getContext('2d');

        function speak(text) {
            subtitles.innerText = text;
            subtitles.style.display = 'block';
            const ut = new SpeechSynthesisUtterance(text);
            ut.onend = () => subtitles.style.display = 'none';
            window.speechSynthesis.speak(ut);
        }

        convertBtn.addEventListener('click', async () => {
            if (!imageInput.files[0]) return alert('Upload an image!');
            const prompt = promptInput.value.toLowerCase();
            
            status.innerText = "Analyzing prompt...";
            convertBtn.disabled = true;
            speak("I'm processing your request: " + prompt);

            const img = new Image();
            img.src = URL.createObjectURL(imageInput.files[0]);
            img.onload = () => {
                canvas.width = img.width; canvas.height = img.height;
                const stream = canvas.captureStream(30);
                const recorder = new MediaRecorder(stream, { mimeType: 'video/webm' });
                const chunks = [];
                recorder.ondataavailable = e => chunks.push(e.data);
                recorder.onstop = () => {
                    video.src = URL.createObjectURL(new Blob(chunks, { type: 'video/webm' }));
                    video.style.display = 'block';
                    convertBtn.disabled = false;
                    speak("Done! Your video is ready.");
                };
                recorder.start();
                
                let start = Date.now();
                const draw = () => {
                    const elapsed = Date.now() - start;
                    const progress = elapsed / 5000;
                    ctx.clearRect(0,0,canvas.width,canvas.height);
                    ctx.save();
                    if(prompt.includes('gray')) ctx.filter = 'grayscale(100%)';
                    if(prompt.includes('zoom')) {
                        let s = 1 + (progress * 0.5);
                        ctx.translate(canvas.width/2, canvas.height/2);
                        ctx.scale(s, s);
                        ctx.translate(-canvas.width/2, -canvas.height/2);
                    }
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    ctx.restore();
                    if(elapsed < 5000) requestAnimationFrame(draw); else recorder.stop();
                };
                draw();
            };
        });
    </script>
